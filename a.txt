    <h1>Amigo Secreto</h1>
    <form id="number-form">
        <p>Digite 5 números para o sorteio:</p>
        <input type="number" id="num1" placeholder="Número 1" required>
        <input type="number" id="num2" placeholder="Número 2" required>
        <input type="number" id="num3" placeholder="Número 3" required>
        <input type="number" id="num4" placeholder="Número 4" required>
        <input type="number" id="num5" placeholder="Número 5" required>
        <button type="submit">Sortear</button>
    </form>

    <div class="result" id="result">
        <h2>Resultado</h2>
        <p>Digite o número para ver quem ele tirou:</p>
        <input type="number" id="query" placeholder="Digite seu número">
        <button onclick="revealPair()">Ver Resultado</button>
        <p id="output"></p>
    </div>

    <script>
        let pairs = {};

        document.getElementById('number-form').addEventListener('submit', function(event) {
            event.preventDefault();

            // Captura os números digitados
            const numbers = [
                document.getElementById('num1').value,
                document.getElementById('num2').value,
                document.getElementById('num3').value,
                document.getElementById('num4').value,
                document.getElementById('num5').value
            ];

            // Validação básica
            if (new Set(numbers).size !== numbers.length) {
                alert('Os números devem ser únicos!');
                return;
            }

            // Embaralha os números para criar os pares
            const shuffled = [...numbers].sort(() => Math.random() - 0.5);

            // Gera os pares (circular, último pega o primeiro)
            pairs = {};
            for (let i = 0; i < numbers.length; i++) {
                pairs[numbers[i]] = shuffled[(i + 1) % numbers.length];
            }

            document.getElementById('result').style.display = 'block';
            alert('Sorteio realizado! Agora digite o número para ver quem ele pegou.');
        });

        function revealPair() {
            const query = document.getElementById('query').value;
            const result = pairs[query];
            if (result) {
                document.getElementById('output').textContent = `Você tirou: ${result}`;
            } else {
                document.getElementById('output').textContent = 'Número não encontrado!';
            }
        }
    </script>
